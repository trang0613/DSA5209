def softmax(scores):
    """Numerically stable row-wise softmax. scores shape: (K,)"""
    s = scores - scores.max()
    e = np.exp(s)
    return e / e.sum()
 
def neg_log_likelihood(Theta, X, y, lam):
    """Full-batch regularised negative log-likelihood (for monitoring)."""
    loss = 0.0
    for xi, yi in zip(X, y):
        p = softmax(Theta @ xi)          # (K,)
        loss -= np.log(p[yi] + 1e-15)
    loss /= len(y)
    loss += lam * np.sum(Theta ** 2)
    return loss
 
def accuracy(Theta, X, y):
    preds = (Theta @ X.T).argmax(axis=0)   # (n,)
    return (preds == y).mean()

def sgd_logistic(X_tr, y_tr, X_te, y_te,
                 lam=1e-4, n_epochs=20,
                 alpha0=0.1, decay=0.01,
                 eval_every=5000, seed=0):
    """
    Stochastic gradient update for one sample (Xi, Yi):
        Theta <- Theta - alpha_k * grad_L(Theta; Xi, Yi)
    grad w.r.t theta_y:
        - xi * (1{y==Yi} - P(y|xi)) + 2*lambda*theta_y
    Step-size schedule: alpha_k = alpha0 / (1 + decay * k)   [Robbins-Monro]
    """
    rng = np.random.default_rng(seed)
    Theta = np.zeros((K, d), dtype=np.float64)   # theta_y are rows
 
    n = len(y_tr)
    train_loss_log, test_acc_log, steps_log = [], [], []
    step = 0
 
    for epoch in range(n_epochs):
        idx = rng.permutation(n)          # shuffle each epoch
        for pos, i in enumerate(idx):
            xi, yi = X_tr[i], y_tr[i]
 
            # forward pass
            scores = Theta @ xi           # (K,)
            p = softmax(scores)           # (K,)
 
            # gradient  (K x d)
            indicator = np.zeros(K)
            indicator[yi] = 1.0
            grad = np.outer(p - indicator, xi) + 2 * lam * Theta
 
            # Robbins-Monro step size
            alpha = alpha0 / (1.0 + decay * step)
 
            Theta -= alpha * grad
            step += 1
 
            if step % eval_every == 0:
                # cheap proxy: NLL on small random subset (5 k samples)
                sub = rng.integers(0, n, size=min(500, n))
                loss = neg_log_likelihood(Theta, X_tr[sub], y_tr[sub], lam)
                tacc = accuracy(Theta, X_te, y_te)
                train_loss_log.append(loss)
                test_acc_log.append(tacc)
                steps_log.append(step)
                print(f"  epoch {epoch+1:2d}  step {step:6d}  "
                      f"NLL≈{loss:.4f}  test-acc={tacc:.4f}")
 
    return Theta, steps_log, train_loss_log, test_acc_log

def gauss_density(y, mu):
    """Returns [f_1(y), f_2(y)] for N(mu_i, 1)."""
    return np.exp(-0.5 * (y - mu) ** 2) / np.sqrt(2 * np.pi)

def forward_backward(Y, A, mu):
    """
    Returns gamma[t,i] = P(Z_t=i | y_{1:T})   (T x 2)
            xi[t,i,j]  = P(Z_t=i, Z_{t+1}=j | y_{1:T})  (T-1 x 2 x 2)
    Uses log-scaling to avoid underflow.
    """
    T = len(Y)
 
    # ── forward pass (log-alpha) ──
    log_alpha = np.zeros((T, 2))
    f0 = gauss_density(Y[0], mu)
    log_alpha[0] = np.log(0.5) + np.log(f0 + 1e-300)
 
    for t in range(1, T):
        ft = gauss_density(Y[t], mu)
        for j in range(2):
            # log sum_i alpha_{t-1}(i) * A_{ij}
            log_sum = np.logaddexp(
                log_alpha[t-1, 0] + np.log(A[0, j] + 1e-300),
                log_alpha[t-1, 1] + np.log(A[1, j] + 1e-300)
            )
            log_alpha[t, j] = np.log(ft[j] + 1e-300) + log_sum
 
    # ── backward pass (log-beta) ──
    log_beta = np.zeros((T, 2))
    # beta_T(i) = 1  =>  log_beta[T-1] = 0
 
    for t in range(T-2, -1, -1):
        ft1 = gauss_density(Y[t+1], mu)
        for i in range(2):
            log_beta[t, i] = np.logaddexp(
                np.log(A[i, 0] + 1e-300) + np.log(ft1[0] + 1e-300) + log_beta[t+1, 0],
                np.log(A[i, 1] + 1e-300) + np.log(ft1[1] + 1e-300) + log_beta[t+1, 1]
            )
 
    # ── gamma ──
    log_gamma = log_alpha + log_beta
    # normalise row-wise
    log_norm  = np.logaddexp(log_gamma[:, 0], log_gamma[:, 1])
    gamma = np.exp(log_gamma - log_norm[:, None])
 
    # ── xi  (T-1 x 2 x 2) ──
    xi = np.zeros((T-1, 2, 2))
    for t in range(T-1):
        ft1 = gauss_density(Y[t+1], mu)
        for i in range(2):
            for j in range(2):
                xi[t, i, j] = np.exp(
                    log_alpha[t, i]
                    + np.log(A[i, j] + 1e-300)
                    + np.log(ft1[j] + 1e-300)
                    + log_beta[t+1, j]
                )
        xi[t] /= xi[t].sum()   # normalise
 
    return gamma, xi

def m_step(Y, gamma, xi):
    """Closed-form M-step updates."""
    # A_{ij} = sum_t xi_t(i,j) / sum_t gamma_t(i)
    A_new = xi.sum(axis=0)                          # (2,2)
    A_new /= A_new.sum(axis=1, keepdims=True)       # row-normalise
 
    # mu_i = sum_t gamma_t(i)*Y_t / sum_t gamma_t(i)
    mu_new = (gamma * Y[:, None]).sum(axis=0) / gamma.sum(axis=0)
    return A_new, mu_new
